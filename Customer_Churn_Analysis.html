#Chart 5 - Services linked to Churn 
services = ['PhoneService', 'InternetService',
            'OnlineSecurity', 'TechSupport']

fig, axes = plt.subplots(2, 2, figsize=(12,8))
fig.suptitle('Services linked to Churn', fontsize=16)

for i, service in enumerate(services):
    sns.countplot(data=df, x=service,
                  hue='Churn',
                  palette=['#66b3ff','#ff6666'],
                  ax=axes[i//2, i%2])
    axes[i//2, i%2].set_title(service)
    axes[i//2, i%2].legend(labels=['Stayed','Churned'])

plt.tight_layout()
plt.show()
